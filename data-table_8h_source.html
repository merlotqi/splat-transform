186  void setValue(size_t index, T value) {
187  if (index >= length()) {
188  if (index >= length()) throw std::out_of_range("Index out of range");
189  }
190 
191  auto visitor = [index, value](auto& vec) {
192  using VectorType = std::decay_t<decltype(vec)>;
193  using Q = typename VectorType::value_type;
194 
195  // Handle non-string input
196  using ValueType = std::decay_t<T>;
197 
198  if constexpr (!std::is_convertible_v<ValueType, Q>) {
199  throw std::runtime_error("Input type cannot be converted to internal column type.");
200  }
201 
202  // Check for potential overflow/truncation
203  if constexpr (std::is_integral_v<Q>) {
204  // Check if value fits in internal type
205  if constexpr (std::is_integral_v<ValueType>) {
206  // Both are integral types
207  if (value > static_cast<ValueType>(std::numeric_limits<Q>::max()) ||
208  value < static_cast<ValueType>(std::numeric_limits<Q>::min())) {
209  throw std::range_error("Value exceeds range of internal integer type.");
210  }
211  } else if constexpr (std::is_floating_point_v<ValueType>) {
212  // Floating point to integer conversion
213  // Check for overflow
214  if (value > static_cast<ValueType>(std::numeric_limits<Q>::max()) ||
215  value < static_cast<ValueType>(std::numeric_limits<Q>::min())) {
216  throw std::range_error("Value exceeds range of internal integer type.");
217  }
218  // Check for truncation
219  if (std::abs(value - std::round(value)) > std::numeric_limits<ValueType>::epsilon() * 10) {
220  throw std::range_error("Floating-point value cannot be exactly represented in internal integer type.");
221  }
222  }
223  } else if constexpr (std::is_floating_point_v<Q>) {
224  // Floating point to floating point conversion
225  if constexpr (std::is_same_v<Q, float> && std::is_same_v<ValueType, double>) {
226  // double to float conversion - check for overflow
227  if (std::abs(value) > std::numeric_limits<float>::max()) {
228  throw std::range_error("Double value exceeds float range.");
229  }
230  }
231  }
232 
233  // Safe to assign
234  vec[index] = static_cast<Q>(value);
235  };

300  bool every(T value) const {
301  return std::visit(
302  [&](const auto& vec) -> bool {
303  using Q = typename std::decay_t<decltype(vec)>::value_type;
304 
305  Q target;
306  if constexpr (std::is_same_v<T, std::string> || std::is_same_v<T, const char*>) {
307  target = parseString<Q>(value);

309  target = static_cast<Q>(value);

314  return std::all_of(vec.begin(), vec.end(), [target, eps](Q x) { return std::abs(x - target) < eps; });
315  } else {

361  template <typename Q, typename T>
362  Q parseString(T value) const {
363  std::string s = value;
364 
365  try {
366  if constexpr (std::is_floating_point_v<Q>) return static_cast<Q>(std::stod(s));
367  if constexpr (std::is_signed_v<Q>) return static_cast<Q>(std::stoll(s));
368  if constexpr (std::is_unsigned_v<Q>) return static_cast<Q>(std::stoull(s));
369  } catch (...) {
370  throw std::runtime_error("Column comparison: String conversion failed");
371  }
372  return Q{};
