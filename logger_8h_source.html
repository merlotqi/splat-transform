51  void logInternal(const char* prefix, const char* file, int line, const char* format, va_list args) {
52  if (level_ == logLevel::silent) {
53  return;
54  }
55 
56  va_list args_copy;
57  va_copy(args_copy, args);
58  int len = std::vsnprintf(nullptr, 0, format, args_copy);
59  va_end(args_copy);
60 
61  if (len < 0) {
62  return;
63  }
64 
65  std::vector<char> buf(len + 1);
66  std::vsnprintf(buf.data(), buf.size(), format, args);
67  std::string formatted_msg(buf.data());
68 
69  std::string_view file_sv(file);
70  size_t last_slash = file_sv.find_last_of("/\\");
71  if (last_slash != std::string_view::npos) {
72  file_sv.remove_prefix(last_slash + 1);
73  }
74 
75  {
76  std::lock_guard<std::mutex> lock(log_mutex_);
77  std::cout << "[" << prefix << "] " << file_sv << ":" << line << " > " << formatted_msg << "\n";
78  std::fflush(stdout);
79  }
80  }
